Integration Snippet


<script>
  const workerUrl = 'YOUR_WORKER_URL';
  const currentUrl = window.location.hostname; // or full URL
  fetch(`${workerUrl}?url=${encodeURIComponent(currentUrl)}`)
    .then(res => res.json())
    .then(data => {
      document.getElementById('view-count').innerText = data.site_views;
      document.getElementById('global-count').innerText = data.total_global_views;
    });
</script>
